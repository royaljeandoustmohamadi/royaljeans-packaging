  m.code,
  m.minStock,

    WHEN mi.type = 'IN' THEN mi.quantity

    ELSE 0

    WHEN COALESCE(SUM(CASE 
      WHEN mi.type = 'IN' THEN mi.quantity 
      WHEN mi.type = 'OUT' THEN -mi.quantity 
      WHEN mi.type = 'ADJUSTMENT' THEN mi.quantity
      ELSE 0

      WHEN mi.type = 'OUT' THEN -mi.quantity 
      WHEN mi.type = 'ADJUSTMENT' THEN mi.quantity
      ELSE 0

    ELSE 'OK'
